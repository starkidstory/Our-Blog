<h2 id="1">C语言计算定时器的参数</h2>
```
#include <stdio.h>
#include <stdlib.h>
#include<windows.h>


int main()
{
double a,b;
while(1)
	{
		int TH1,TL1;
     		scanf("%lf",&a);//毫秒
    		scanf("%lf",&b);//晶振频率 
    	 	TH1=(int)(65536-1000*a*b/12)/256;
    		TL1=(int)(65536-1000*a*b/12)%256;
    		printf("TH1:%d\nTL1:%d",TH1,TL1);
    		Sleep(5000);  
    		system("cls");   
    	 }
system("pause");
return 0;
}
```
